', unsafe_allow_html=True) # ================= LOGIC (THE FIX) ================= def parse_invoice_text(text_content): """ دالة لمعالجة نص الفاتورة واستخراج البيانات مع تجاهل تكرار الهيدر """ records = [] # 1. نستخرج رقم الحساب ورقم الفاتورة من الهيدر لاستبعادهما لاحقاً # نبحث عن النمط المتكرر في الهيدر (مثال: 6.225719 2607010130093682 322 2026/07/01) invoice_header_pattern = re.compile(r'(\d+\.\d+)\s+(\d{10,})\s+322\s+\d{4}/\d{2}/\d{2}') header_matches = invoice_header_pattern.findall(text_content) # نجمع الأرقام التي تظهر في الهيدر لنستبعدها من المعالجة excluded_numbers = set() for match in header_matches: excluded_numbers.add(match[0]) # رقم الحساب excluded_numbers.add(match[1]) # رقم الفاتورة # 2. تقسيم النص إلى أسطر ومعالجتها lines = text_content.split('\n') for line in lines: line = line.strip() if not line: continue # البحث عن أرقام الهواتف المصرية فقط (010, 011, 012, 015) phone_match = re.search(r'(01[0125]\d{8})', line) if phone_match: phone_number = phone_match.group(1) # === الحل الجذري: استبعاد أرقام الهيدر === if phone_number in excluded_numbers: continue # ====================================== # نفترض أن القيم تأتي بعد رقم الهاتف أو في نفس السطر # نقوم بتنظيف السطر من رقم الهاتف لاستخراج الأرقام الأخرى بدقة clean_line = line.replace(phone_number, "").strip() # استخراج الأرقام العشرية والصحيحة values = re.findall(r'[-+]?\d*\.?\d+', clean_line) # تحويل القيم إلى أرقام عائمة numeric_values = [] for v in values: try: numeric_values.append(float(v)) except ValueError: continue # التأكد من وجود بيانات كافية (على الأقل الرسوم والضرائب والإجمالي) if len(numeric_values) >= 4: record = { "رقم المحمول": phone_number, "البيانات_الخام": numeric_values # يمكن تفصيلها لاحقاً حسب ترتيب الأعمدة في ملفك الفعلي } # محاولة تعيين الأعمدة بناءً على الترتيب الشائع في فواتير اتصالات # عادة الترتيب: رسوم شهرية | خدمات | ضريبة | إجمالي # يمكنك تعديل المؤشرات [0], [1]... حسب ترتيب الأعمدة في ملفك الفعلي if len(numeric_values) >= 4: record["رسوم شهرية"] = numeric_values[0] record["خدمات"] = numeric_values[1] if len(numeric_values) > 1 else 0 record["ضرائب"] = numeric_values[-2] # قبل الأخير غالباً ضريبة record["الإجمالي"] = numeric_values[-1] # الأخير غالباً الإجمالي records.append(record) return pd.DataFrame(records) # ================= UI ================= uploaded_file = st.file_uploader(" رفع ملف النص (TXT)", type=["txt"]) if uploaded_file is not None: # قراءة محتوى الملف كنص stringio = io.StringIO(uploaded_file.getvalue().decode("utf-8")) text_data = stringio.read() # معالجة البيانات df_result = parse_invoice_text(text_data) if not df_result.empty: st.success(f"تم استخراج {len(df_result)} سجل بنجاح!") st.dataframe(df_result, use_container_width=True) # تحميل Excel buf = io.BytesIO() with pd.ExcelWriter(buf, engine="openpyxl") as writer: df_result.to_excel(writer, index=False) st.download_button("📥 تحميل Excel", data=buf.getvalue(), file_name="Telecom_Report.xlsx") else: st.warning("لم يتم العثور على بيانات صالحة في الملف.")
