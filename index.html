import re import pandas as pd import io def parse_invoice_text(text_content): """ دالة لمعالجة نص الفاتورة واستخراج البيانات مع تجاهل تكرار الهيدر """ records = [] # 1. استخراج رقم الفاتورة ورقم الحساب من الهيدر لاستبعادهما لاحقاً # نبحث عن النمط المتكرر في بداية كل صفحة invoice_header_pattern = re.compile(r'(\d{10,})\s+(\d{10,})\s+322\s+2026/07/01') header_matches = invoice_header_pattern.findall(text_content) # نجمع الأرقام التي تظهر في الهيدر لنستبعدها من البيانات excluded_numbers = set() for match in header_matches: excluded_numbers.add(match[0]) # رقم الحساب excluded_numbers.add(match[1]) # رقم الفاتورة # 2. تقسيم النص إلى أسطر ومعالجتها lines = text_content.split('\n') for line in lines: line = line.strip() if not line: continue # البحث عن أرقام الهواتف المصرية فقط (010, 011, 012, 015) phone_match = re.search(r'(01[0125]\d{8})', line) if phone_match: phone_number = phone_match.group(1) # === الحل الجذري: استبعاد أرقام الهيدر === if phone_number in excluded_numbers: continue # ====================================== # استخراج القيم الرقمية من السطر # نفترض أن القيم تأتي بعد رقم الهاتف أو في نفس السطر # نقوم بتنظيف السطر من رقم الهاتف لاستخراج الأرقام الأخرى بدقة clean_line = line.replace(phone_number, '').strip() # استخراج الأرقام العشرية والصحيحة values = re.findall(r'[-+]?\d*\.?\d+', clean_line) # تحويل القيم إلى أرقام عائمة numeric_values = [] for v in values: try: numeric_values.append(float(v)) except ValueError: continue # التأكد من وجود بيانات كافية (على الأقل الرسوم والضرائب والإجمالي) # في العادة الفواتير تحتوي على: رسوم شهرية، خدمات، ضريبة، إجمالي if len(numeric_values) >= 4: record = { "رقم المحمول": phone_number, "البيانات_الخام": numeric_values # يمكن تفصيلها لاحقاً حسب ترتيب الأعمدة } # محاولة تعيين الأعمدة بناءً على الترتيب الشائع في فواتير اتصالات # عادة الترتيب: رسوم شهرية | خدمات | ضريبة | إجمالي # يمكنك تعديل المؤشرات [0], [1]... حسب ترتيب الأعمدة في ملفك الفعلي if len(numeric_values) >= 4: record["رسوم شهرية"] = numeric_values[0] record["خدمات"] = numeric_values[1] if len(numeric_values) > 1 else 0 record["ضرائب"] = numeric_values[-2] # قبل الأخير غالباً ضريبة record["الإجمالي"] = numeric_values[-1] # الأخير غالباً الإجمالي records.append(record) return pd.DataFrame(records) # ========================================================= # مثال على الاستخدام (يمكنك استبدال النص بمحتوى ملفك الفعلي) # ========================================================= # محاكاة لنص يحتوي على تكرار الهيدر كما في مشكلتك sample_text = """ مول 01013009368 2607010130093682 322 2026/07/01 | ﻲﻟﺎﻤﺟﺇ | ﻤﻴ ﺐﺋﺍﺮﻀﻟﺍ | ... | ---|---|---| | ﻝﻮﻤﺤﻣ 01148267883 ... | 375 | 0 | ... | 86.70 | 461.7 | 01013009368 2607010130093682 322 2026/07/01 | ﻲﻟﺎﻤﺟﺇ | ﺔﻤﻴﻗ ﺐﺋﺍﺮﻀﻟﺍ | ... | ---|---|---| | ﻝﻮﻤﺤﻣ 01148218760 ... | 375 | 0 | ... | 86.70 | 461.7 | 01013009368 2607010130093682 322 2026/07/01 """ # تشغيل الدالة df_result = parse_invoice_text(sample_text) # عرض النتيجة للتأكد من عدم تكرار رقم الهيدر print("عدد الصفوف المستخرجة:", len(df_result)) print(df_result)
