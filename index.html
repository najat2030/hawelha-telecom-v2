def parse_file(file, is_arabic): records = [] try: file_bytes = file.read() doc = fitz.open(stream=file_bytes, filetype="pdf") # نبدأ من الصفحة الثالثة كما هو معتاد في هذه الفواتير for page_num in range(2, len(doc)): page = doc[page_num] tabs = page.find_tables() if not tabs: continue for table in tabs: extracted_rows = table.extract() if not extracted_rows: continue # نمر على الصفوف للبحث عن سطر يحتوي على "محمول" ورقم هاتف for i, row in enumerate(extracted_rows): if not row: continue # تحويل الصف لنص للبحث عنه row_text = " ".join([str(c) for c in row if c]) # البحث عن نمط "محمول" متبوع برقم هاتف phone_match = re.search(r'محمول\s*(01[0125]\d{8})', row_text) if phone_match: p = phone_match.group(1) # التحقق من وجود سطر بيانات مالي بعده if i + 1 < len(extracted_rows): next_row = extracted_rows[i+1] # استخراج الأرقام من سطر البيانات المالي # نقوم بتنظيف الخلايا وتحويلها لأرقام financial_vals = [] for cell in next_row: if cell: try: # تنظيف النص من أي رموز غير رقمية ما عدا النقطة والناقص clean_val = str(cell).replace(',', '').strip() if re.match(r'^-?\d+(\.\d+)?$', clean_val): financial_vals.append(float(clean_val)) except: pass # الآن لدينا قائمة بالأرقام المالية بالترتيب # بناءً على الصورة، الترتيب المتوقع هو: # [رسوم شهرية, خدمات, محلي..., دولي..., تجوال..., تسويات, ضرائب, إجمالي] # لكن قد تختلف عدد الأصفار، لذا نعتمد على المواقع الثابتة للأعمدة المهمة # في العادة: # index 0: رسوم شهرية # ... (أصفار أو قيم استهلاك) ... # index -2: الضرائب # index -1: الإجمالي if len(financial_vals) >= 2: # تحديد القيم بناءً على مواقعها النسبية monthly_fee = financial_vals[0] if len(financial_vals) > 0 else 0 taxes = financial_vals[-2] if len(financial_vals) >= 2 else 0 total = financial_vals[-1] if len(financial_vals) >= 1 else 0 # محاولة استخراج باقي القيم إذا كانت موجودة بشكل واضح # نفترض أن الأعمدة الوسطى (استهلاك) تبدأ من index 1 # لكن لضمان الدقة، سنضع الأصفار كقيمة افتراضية للأعمدة التي قد تكون فارغة في النص record = { "محمول": p, "رسوم شهرية": monthly_fee, "رسوم الخدمات": financial_vals[1] if len(financial_vals) > 1 else 0, "مكالمات محلية": 0, # يتم تحديدها بدقة إذا كان هناك تمييز في الخلايا "رسائل محلية": 0, "إنترنت محلية": 0, "مكالمات دولية": 0, "رسائل دولية": 0, "مكالمات تجوال": 0, "رسائل تجوال": 0, "إنترنت تجوال": 0, "رسوم تسويات": 0, "ضرائب": taxes, "إجمالي": total } # تحسين: إذا كان عدد القيم أكبر، نحاول توزيعها # لكن safest way هو الاعتماد على الأول والأخير والثاني قبل الأخير # لأن الأعمدة الوسطى غالباً ما تكون مدمجة أو أصفار في هذا النوع من الفواتير records.append(record) doc.close() except Exception as e: st.error(f"حدث خطأ أثناء معالجة الملف: {str(e)}") return records
