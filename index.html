', unsafe_allow_html=True) # ================= LOGIC ================= def normalize(t): return (t or "").replace("−", "-").replace("–", "-").replace("—", "-") def extract_numbers(text): if not text: return [] text = normalize(str(text)) text = re.sub(r'\((\d+\.?\d*)\)', r'-\1', text) text = re.sub(r'(\d+\.?\d*)-', r'-\1', text) text = re.sub(r'-\s+(\d)', r'-\1', text) numbers = re.findall(r'-?\d+(?:\.\d+)?', text) return [float(n) for n in numbers] def parse_file(file, is_arabic): records = [] try: file_bytes = file.read() doc = fitz.open(stream=file_bytes, filetype="pdf") for page_num in range(2, len(doc)): page = doc[page_num] tabs = page.find_tables() if not tabs: continue for table in tabs: extracted_rows = table.extract() if not extracted_rows: continue for i, row in enumerate(extracted_rows): if not row: continue # دمج خلايا الصف في نص واحد للبحث row_text = " ".join([str(c) for c in row if c]) # البحث عن كلمة "محمول" أولاً if "محمول" in row_text: # استخراج أول رقم موبايل يظهر بعد كلمة محمول phone_match = re.search(r'محمول\s*(01[0125]\d{8})', row_text) if phone_match: p = phone_match.group(1) # استخراج الأرقام من باقي خلايا الصف (باستثناء الخلية التي تحتوي على الموبايل) # نجمع الخلايا التي لا تحتوي على كلمة "محمول" ولا تحتوي على الرقم نفسه numeric_cells_text = [] for cell in row: cell_str = str(cell) if "محمول" not in cell_str and p not in cell_str: numeric_cells_text.append(cell_str) combined_nums_text = " ".join(numeric_cells_text) vals = extract_numbers(combined_nums_text) # إذا لم نجد أرقام كافية في نفس الصف، نحاول الصف التالي (كما في المنطق السابق) if len(vals) < 13 and i + 1 < len(extracted_rows): next_row = extracted_rows[i+1] next_text = " ".join([str(c) for c in next_row if c]) # نتأكد أن الصف التالي ليس سطراً جديداً لعميل آخر if "محمول" not in next_text: nxt_vals = extract_numbers(next_text) if len(nxt_vals) > len(vals): vals = nxt_vals if len(vals) >= 13: # بناء السجل باستخدام أول 13 رقماً تم العثور عليها def build_record(v): def g(idx): return v[idx] if idx < len(v) else 0 return { "محمول": p, "رسوم شهرية": g(0), "رسوم الخدمات": g(1), "مكالمات محلية": g(2), "رسائل محلية": g(3), "إنترنت محلية": g(4), "مكالمات دولية": g(5), "رسائل دولية": g(6), "مكالمات تجوال": g(7), "رسائل تجوال": g(8), "إنترنت تجوال": g(9), "رسوم تسويات": g(10), "ضرائب": g(11), "إجمالي": g(12) } # إضافة السجل مباشرة بدون تعقيدات الـ Scoring الزائدة # لأننا الآن متأكدون من ترتيب الأرقام بفضل ربطها بكلمة "محمول" rec = build_record(vals[:13]) records.append(rec) doc.close() except Exception as e: st.error(f"حدث خطأ أثناء معالجة الملف: {str(e)}") return records # ================= UI ================= files = st.file_uploader("📂 رفع ملفات PDF", type=["pdf"], accept_multiple_files=True) mode = st.radio("إعدادات اللغة", ["Auto 🤖", "عربي 🇪", "English 🇺🇸"], horizontal=True) if st.button("🚀 بدء المعالجة والتحليل"): if files: progress_bar = st.progress(0) all_data = [] for idx, file in enumerate(files): file.seek(0) if mode == "Auto 🤖": try: file_bytes_temp = file.read() temp_doc = fitz.open(stream=file_bytes_temp, filetype="pdf") txt = temp_doc[0].get_text() or "" temp_doc.close() is_ar = True if re.search(r'[\u0600-\u06FF]', txt) else False except: is_ar = True else: is_ar = True if mode == "عربي 🇪🇬" else False file.seek(0) data = parse_file(file, is_ar) all_data.extend(data) progress_bar.progress((idx + 1) / len(files)) gc.collect() if all_data: df = pd.DataFrame(all_data) st.markdown("### 📈 ملخص التحليل المالي") m1, m2, m3, m4, m5 = st.columns(5) with m1: st.markdown(f'
