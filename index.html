def parse_file(file, is_arabic): records = [] try: file_bytes = file.read() doc = fitz.open(stream=file_bytes, filetype="pdf") for page_num in range(2, len(doc)): page = doc[page_num] tabs = page.find_tables() if not tabs: continue for table in tabs: extracted_rows = table.extract() if not extracted_rows: continue for i, row in enumerate(extracted_rows): if not row: continue # 1. دمج خلايا الصف في نص واحد للبحث row_text = " ".join([str(c) for c in row if c]) # 2. البحث عن كلمة "محمول" واستخراج الرقم اللي بعدها مباشرة phone_match = re.search(r'محمول\s*(01[0125]\d{8})', row_text) if phone_match: p = phone_match.group(1) # ده الرقم الصح اللي جنب محمول # استخراج باقي الأرقام من الخلايا اللي مفيهاش "محمول" ولا رقم التليفون numeric_cells_text = [] for cell in row: cell_str = str(cell) if "محمول" not in cell_str and p not in cell_str: numeric_cells_text.append(cell_str) combined_nums_text = " ".join(numeric_cells_text) vals = extract_numbers(combined_nums_text) # لو الأرقام قليلة، نحاول نجيب الباقي من الصف اللي بعده (لو مش عميل تاني) if len(vals) < 13 and i + 1 < len(extracted_rows): next_row = extracted_rows[i+1] next_text = " ".join([str(c) for c in next_row if c]) if "محمول" not in next_text: # تأكد إنه مش سطر عميل جديد nxt_vals = extract_numbers(next_text) if len(nxt_vals) > len(vals): vals = nxt_vals if len(vals) >= 13: # بناء السجل باستخدام أول 13 رقم تم العثور عليها def build_record(v): def g(idx): return v[idx] if idx < len(v) else 0 return { "محمول": p, "رسوم شهرية": g(0), "رسوم الخدمات": g(1), "مكالمات محلية": g(2), "رسائل محلية": g(3), "إنترنت محلية": g(4), "مكالمات دولية": g(5), "رسائل دولية": g(6), "مكالمات تجوال": g(7), "رسائل تجوال": g(8), "إنترنت تجوال": g(9), "رسوم تسويات": g(10), "ضرائب": g(11), "إجمالي": g(12) } # إضافة السجل مباشرة لأننا متأكدون من ترتيب الأرقام بفضل ربطها بكلمة "محمول" rec = build_record(vals[:13]) records.append(rec) except Exception as e: st.error(f"حدث خطأ أثناء معالجة الملف: {str(e)}") return records
